DECLARE
new_sal EMP3.salary %type;
incentive number(6,2);
BEGIN
select salary into new_sal from EMP3 where emp_id=102;
incentive := new_sal*0.1;
dbms_output.put_line('Incentive of 110: ' || incentive);
EXCEPTION
WHEN NO_DATA_FOUND then 
dbms_output.put_line('110 is not available');
END;
/
